FROM

SELECT
P.NOMBRE,
P.EMAIL,
C.NOMBRE_CIUDAD,
I.NOMBRE_PAIS 
FROM 
PERSONAS P,CIUDAD C,PAIS I 
WHERE
P.ID_CIUDAD=C.ID_CIUDAD AND
C.ID_PAIS=I.ID_PAIS AND
C.NOMBRE_CIUDAD!='Medellin' AND
C.NOMBRE_CIUDAD!='Bogota' AND
P.EMAIL LIKE '%.com%'
;
